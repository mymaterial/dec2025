postgres: walwriter

	- writes information from wal buffers to wal files
	- on every commit
	
postgres: background writer

	- writes information from shared buffers to data files
	- bgwriter_delay = 200ms (24*7 with a delay of 200ms)
	- bgwriter_lru_maxpages = 100 (100 blocks at a time)
	
postgres: checkpointer

	- ensure your dirty buffers/shared buffers data is written to datafiles
	- checkpoint_timeout = 5min (crosschecks/occurs for every 5 min)

postgres: logger

	- writes events happening in database to alert log file.
